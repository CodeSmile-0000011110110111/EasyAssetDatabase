185 public static Boolean IsValid(String path)
186 {
187 var isValid = true;
188
189 try
190 {
191 // System.IO will throw for most illegal chars, plus some extra checks
192 var fileName = System.IO.Path.GetFileName(path);
193 var folderName = System.IO.Path.GetDirectoryName(path);
194
195 // check folder name for some chars that System.IO allows in GetDirectoryName
196 var testIllegalChars = new Func<Char, Boolean>(c => c == '*' || c == '?' || c == ':');
197 isValid = folderName.Any(testIllegalChars) == false;
198
199 if (isValid)
200 {
201 // check filename for some chars that System.IO allows in GetFileName
202 fileName = path.Substring(folderName.Length, path.Length - folderName.Length);
203 isValid = fileName.Any(testIllegalChars) == false;
204 }
205 }
206 catch (Exception ex)
207 {
208 SetLastErrorMessage($"{ex.Message} => \"{path}\"");
209 isValid = false;
210 }
211
212 return isValid;
213 }

245 public static GUID CreateFolders(Path path)
246 {
247 ThrowIf.ArgumentIsNull(path, nameof(path));
248 ThrowIf.PathIsNotValid(path);
249
250 if (FileExists(path))
251 return path.FolderPath.Guid;
252
253 // if the last part has an extension we assume the path points to a file
254 var isPresumablyFilePath = String.IsNullOrEmpty(path.Extension) == false;
255 var folderPath = isPresumablyFilePath ? path.FolderPath : path;
256 if (FolderExists(folderPath))
257 return folderPath.Guid;
258
259 var folderNames = ((String)folderPath).Split(new[] { '/' });
260 var folderGuid = GuidForExistingPath(folderNames[0]); // first is "Assets"
261 var partialPath = folderNames[0];
262 for (var i = 1; i < folderNames.Length; i++)
263 {
264 partialPath += $"/{folderNames[i]}";
265 if (FolderExists(partialPath))
266 {
267 folderGuid = GuidForExistingPath(partialPath);
268 continue;
269 }
270
271 folderGuid = CreateSubFolder(Get(folderGuid), folderNames[i]);
272 }
273
274 return folderGuid;
275 }

323 private static String ToRelative(String fullOrRelativePath)
324 {
325 var relativePath = fullOrRelativePath;
326 if (IsRelative(relativePath) == false)
327 {
328 ThrowIf.NotAProjectPath(fullOrRelativePath);
329 relativePath = MakeRelative(fullOrRelativePath);
330 }

334 ThrowIf.PathIsNotValid(relativePath);
335 return relativePath;
336 }

338 private static Boolean IsRelative(String path)
339 {
340 path = path.TrimStart('/').ToLower();
341
342 // path must start with given project root subfolder names (eg 'Assets', 'Packages', 'Library' ..)
343 // and bei either just the subfolder (length equals) or be followed by a path separator
344 foreach (var allowedSubfolder in s_AllowedAssetSubfolders)
345 {
346 var doesStartsWith = path.StartsWith(allowedSubfolder);
347 var subfolderLength = allowedSubfolder.Length;
348 var lengthMatches = path.Length == subfolderLength;
349 if (doesStartsWith && (lengthMatches || path[subfolderLength].Equals('/')))
350 return true;
351 }
352
353 return false;
354 }
