644 internal static Boolean CopyInternal(Path sourcePath, Path destinationPath, Boolean overwriteExisting)
645 {
646 ThrowIf.ArgumentIsNull(sourcePath, nameof(sourcePath));
647 ThrowIf.ArgumentIsNull(destinationPath, nameof(destinationPath));
648 ThrowIf.AssetPathNotInDatabase(sourcePath);
649 ThrowIf.SourceAndDestPathAreEqual(sourcePath, destinationPath);
650
651 destinationPath.CreateFolders();
652
653 var success = AssetDatabase.CopyAsset(sourcePath, destinationPath);
654 SetLastErrorMessage(success ? String.Empty : $"failed to copy {sourcePath} to {destinationPath}");
655 return success;
656 }
