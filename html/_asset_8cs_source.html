595 private void InitWithPath(Path path)
596 {
597 ThrowIf.ArgumentIsNull(path, nameof(path));
598 ThrowIf.DoesNotExistInFileSystem(path);
599
600 m_AssetPath = path;
601 m_MainObject = Status.IsImported(path) ? LoadMain<Object>() : File.ImportAndLoad<Object>(path);
602
603 ThrowIf.AssetLoadReturnedNull(m_MainObject, m_AssetPath);
604 }
605
606 private void InitWithMainObject(Object mainObject)
607 {
608 ThrowIf.ArgumentIsNull(mainObject, nameof(mainObject));
609 ThrowIf.NotInDatabase(mainObject);
610
611 m_MainObject = mainObject;
612 m_AssetPath = Path.Get(mainObject);
613 }
