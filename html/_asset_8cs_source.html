566 private void InitWithPath(Path path)
567 {
568 ThrowIf.ArgumentIsNull(path, nameof(path));
569 ThrowIf.DoesNotExistInFileSystem(path);
570
571 m_AssetPath = path;
572 m_MainObject = Status.IsImported(path) ? LoadMain<Object>() : File.ImportAndLoad<Object>(path);
573
574 ThrowIf.AssetLoadReturnedNull(m_MainObject, m_AssetPath);
575 }
576
577 private void InitWithMainObject(Object mainObject)
578 {
579 ThrowIf.ArgumentIsNull(mainObject, nameof(mainObject));
580 ThrowIf.NotInDatabase(mainObject);
581
582 m_MainObject = mainObject;
583 m_AssetPath = Path.Get(mainObject);
584 }
