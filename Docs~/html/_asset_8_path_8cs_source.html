207 public static Boolean IsValid(String path)
208 {
209 var isValid = true;
210
211 try
212 {
213 // System.IO will throw for most illegal chars, plus some extra checks
214 var fileName = System.IO.Path.GetFileName(path);
215 var folderName = System.IO.Path.GetDirectoryName(path);
216
217 // check folder name for some chars that System.IO allows in GetDirectoryName
218 var testIllegalChars = new Func<Char, Boolean>(c => c == '*' || c == '?' || c == ':');
219 isValid = folderName.Any(testIllegalChars) == false;
220
221 if (isValid)
222 {
223 // check filename for some chars that System.IO allows in GetFileName
224 fileName = path.Substring(folderName.Length, path.Length - folderName.Length);
225 isValid = fileName.Any(testIllegalChars) == false;
226 }
227 }
228 catch (Exception ex)
229 {
230 SetLastErrorMessage($"{ex.Message} => \"{path}\"");
231 isValid = false;
232 }
233
234 return isValid;
235 }

258 public static GUID CreateFolders(Path path)
259 {
260 ThrowIf.ArgumentIsNull(path, nameof(path));
261 ThrowIf.PathIsNotValid(path);
262
263 if (FileExists(path))
264 return path.FolderPath.Guid;
265
266 // if the last part has an extension we assume the path points to a file
267 var isPresumablyFilePath = String.IsNullOrEmpty(path.Extension) == false;
268 var folderPath = isPresumablyFilePath ? path.FolderPath : path;
269 if (FolderExists(folderPath))
270 return folderPath.Guid;
271
272 var folderNames = ((String)folderPath).Split(new[] { '/' });
273 var folderGuid = GuidForExistingPath(folderNames[0]); // first is "Assets"
274 var partialPath = folderNames[0];
275 for (var i = 1; i < folderNames.Length; i++)
276 {
277 partialPath += $"/{folderNames[i]}";
278 if (FolderExists(partialPath))
279 {
280 folderGuid = GuidForExistingPath(partialPath);
281 continue;
282 }
283
284 folderGuid = CreateSubFolder(Get(folderGuid), folderNames[i]);
285 }
286
287 return folderGuid;
288 }

359 private static Boolean IsRelative(String path)
360 {
361 path = path.TrimStart('/').ToLower();
362
363 // path must start with given project root subfolder names (eg 'Assets', 'Packages', 'Library' ..)
364 // and bei either just the subfolder (length equals) or be followed by a path separator
365 foreach (var allowedSubfolder in s_AllowedAssetSubfolders)
366 {
367 var doesStartsWith = path.StartsWith(allowedSubfolder);
368 var subfolderLength = allowedSubfolder.Length;
369 var lengthMatches = path.Length == subfolderLength;
370 if (doesStartsWith && (lengthMatches || path[subfolderLength].Equals('/')))
371 return true;
372 }
373
374 return false;
375 }
