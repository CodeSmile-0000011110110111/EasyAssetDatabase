563 private void InitWithPath(Path path)
564 {
565 ThrowIf.ArgumentIsNull(path, nameof(path));
566 ThrowIf.DoesNotExistInFileSystem(path);
567
568 m_AssetPath = path;
569 m_MainObject = Status.IsImported(path) ? LoadMain<Object>() : File.ImportAndLoad<Object>(path);
570
571 ThrowIf.AssetLoadReturnedNull(m_MainObject, m_AssetPath);
572 }
573
574 private void InitWithMainObject(Object mainObject)
575 {
576 ThrowIf.ArgumentIsNull(mainObject, nameof(mainObject));
577 ThrowIf.NotInDatabase(mainObject);
578
579 m_MainObject = mainObject;
580 m_AssetPath = Path.Get(mainObject);
581 }
