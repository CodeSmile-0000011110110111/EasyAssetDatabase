528 private void InitWithPath(Path path)
529 {
530 ThrowIf.ArgumentIsNull(path, nameof(path));
531 ThrowIf.DoesNotExistInFileSystem(path);
532
533 m_AssetPath = path;
534 m_MainObject = Status.IsImported(path) ? LoadMain<Object>() : File.ImportAndLoad<Object>(path);
535
536 ThrowIf.AssetLoadReturnedNull(m_MainObject, m_AssetPath);
537 }

539 private void InitWithObject(Object obj)
540 {
541 ThrowIf.ArgumentIsNull(obj, nameof(obj));
542 ThrowIf.NotInDatabase(obj);
543
544 m_MainObject = obj;
545 m_AssetPath = Path.Get(obj);
546 }
