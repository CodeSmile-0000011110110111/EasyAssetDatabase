165 public static ValueTuple<GUID, Int64> GetGuidAndFileId(Object obj)
166 {
167 if (obj == null)
168 return (new GUID(), 0L);
169
170 // explicit variable + assign because Unity 2021 has both long and int variants of the TryGetGUID* method
171 var localId = Int64.MaxValue;
172 return AssetDatabase.TryGetGUIDAndLocalFileIdentifier(obj, out var guid, out localId)
