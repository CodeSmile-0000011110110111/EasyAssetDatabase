103 public static Boolean IsValid(String path)
104 {
105 var isValid = true;
106
107 try
108 {
109 // System.IO will throw for most illegal chars, plus some extra checks
110 var fileName = System.IO.Path.GetFileName(path);
111 var folderName = System.IO.Path.GetDirectoryName(path);
112
113 // check folder name for some chars that System.IO allows in GetDirectoryName
114 var testIllegalChars = new Func<Char, Boolean>(c => c == '*' || c == '?' || c == ':');
115 isValid = folderName.Any(testIllegalChars) == false;
116
117 if (isValid)
118 {
119 // check filename for some chars that System.IO allows in GetFileName
120 fileName = path.Substring(folderName.Length, path.Length - folderName.Length);
121 isValid = fileName.Any(testIllegalChars) == false;
122 }
123 }
124 catch (Exception ex)
125 {
126 SetLastErrorMessage($"{ex.Message} => \"{path}\"");
127 isValid = false;
128 }
129
130 return isValid;
131 }

154 public static GUID CreateFolders(Path path)
155 {
156 ThrowIf.ArgumentIsNull(path, nameof(path));
157 ThrowIf.PathIsNotValid(path);
158
159 if (FileExists(path))
160 return path.FolderPath.Guid;
161
162 // if the last part has an extension we assume the path points to a file
163 var isPresumablyFilePath = String.IsNullOrEmpty(path.Extension) == false;
164 var folderPath = isPresumablyFilePath ? path.FolderPath : path;
165 if (FolderExists(folderPath))
166 return folderPath.Guid;
167
168 var folderNames = ((String)folderPath).Split(new[] { '/' });
169 var folderGuid = GuidForExistingPath(folderNames[0]); // first is "Assets"
170 var partialPath = folderNames[0];
171 for (var i = 1; i < folderNames.Length; i++)
172 {
173 partialPath += $"/{folderNames[i]}";
174 if (FolderExists(partialPath))
175 {
176 folderGuid = GuidForExistingPath(partialPath);
177 continue;
178 }
179
180 folderGuid = CreateSubFolder(Get(folderGuid), folderNames[i]);
181 }
182
183 return folderGuid;
184 }

255 private static Boolean IsRelative(String path)
256 {
257 path = path.TrimStart('/').ToLower();
258
259 // path must start with given project root subfolder names (eg 'Assets', 'Packages', 'Library' ..)
260 // and bei either just the subfolder (length equals) or be followed by a path separator
261 foreach (var allowedSubfolder in s_AllowedAssetSubfolders)
262 {
263 var doesStartsWith = path.StartsWith(allowedSubfolder);
264 var subfolderLength = allowedSubfolder.Length;
265 var lengthMatches = path.Length == subfolderLength;
266 if (doesStartsWith && (lengthMatches || path[subfolderLength].Equals('/')))
267 return true;
268 }
269
270 return false;
271 }
272

274 fullOrRelativePath.Substring(FullProjectPath.Length).Trim('/');
