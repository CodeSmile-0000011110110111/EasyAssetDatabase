876 internal static Object CreateInternal([NotNull] Byte[] bytes, [NotNull] Path path)
877 {
878 ThrowIf.ArgumentIsNull(bytes, nameof(bytes));
879 ThrowIf.ArgumentIsNull(path, nameof(path));
880
881 path.CreateFolders();
882 System.IO.File.WriteAllBytes(path, bytes);
883 return ImportAndLoad<Object>(path);
884 }

886 internal static Object CreateInternal([NotNull] String contents, [NotNull] Path path)
887 {
888 ThrowIf.ArgumentIsNull(contents, nameof(contents));
889 ThrowIf.ArgumentIsNull(path, nameof(path));
890
891 path.CreateFolders();
892 System.IO.File.WriteAllText(path, contents, Encoding.UTF8); // string assets ought to be UTF8
893 return ImportAndLoad<Object>(path);
894 }

906 internal static Boolean CopyInternal([NotNull] Path sourcePath, [NotNull] Path destinationPath,
907 Boolean overwriteExisting)
908 {
909 ThrowIf.ArgumentIsNull(sourcePath, nameof(sourcePath));
910 ThrowIf.ArgumentIsNull(destinationPath, nameof(destinationPath));
911 ThrowIf.AssetPathNotInDatabase(sourcePath);
912 ThrowIf.SourceAndDestPathAreEqual(sourcePath, destinationPath);
913
914 destinationPath.CreateFolders();
915
916 var success = AssetDatabase.CopyAsset(sourcePath, destinationPath);
917 SetLastErrorMessage(success ? String.Empty : $"failed to copy {sourcePath} to {destinationPath}");
918 return success;
919 }
920
921 private static void SaveInternal([NotNull] Object asset, Boolean forceSave = false)
922 {
923 ThrowIf.ArgumentIsNull(asset, nameof(asset));
924 ThrowIf.NotInDatabase(asset);
925
926 if (forceSave)
927 EditorUtility.SetDirty(asset);
928
929 AssetDatabase.SaveAssetIfDirty(asset);
930 }
