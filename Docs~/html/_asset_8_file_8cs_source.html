648 internal static Boolean CopyInternal([NotNull] Path sourcePath, [NotNull] Path destinationPath,
649 Boolean overwriteExisting)
650 {
651 ThrowIf.ArgumentIsNull(sourcePath, nameof(sourcePath));
652 ThrowIf.ArgumentIsNull(destinationPath, nameof(destinationPath));
653 ThrowIf.AssetPathNotInDatabase(sourcePath);
654 ThrowIf.SourceAndDestPathAreEqual(sourcePath, destinationPath);
655
656 destinationPath.CreateFolders();
657
658 var success = AssetDatabase.CopyAsset(sourcePath, destinationPath);
659 SetLastErrorMessage(success ? String.Empty : $"failed to copy {sourcePath} to {destinationPath}");
660 return success;
661 }
