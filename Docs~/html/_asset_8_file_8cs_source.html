448 public static Boolean Copy(Path sourcePath, Path destinationPath, Boolean overwriteExisting = false)
449 {
450 ThrowIf.ArgumentIsNull(sourcePath, nameof(sourcePath));
451 ThrowIf.ArgumentIsNull(destinationPath, nameof(destinationPath));
452 ThrowIf.AssetPathNotInDatabase(sourcePath);
453 ThrowIf.OverwritingSamePath(sourcePath, destinationPath, overwriteExisting);
454
455 var newDestPath = Path.GetOverwriteOrUnique(destinationPath, overwriteExisting);
456 newDestPath.CreateFolders();
457
458 var success = AssetDatabase.CopyAsset(sourcePath, newDestPath);
459 SetLastErrorMessage(success ? String.Empty : $"failed to copy {sourcePath} to {newDestPath}");
460 return success;
461 }

660 public static void BatchEditing(Action massAssetFileEditAction, Boolean rethrowExceptions = false)
661 {
662 ThrowIf.ArgumentIsNull(massAssetFileEditAction, nameof(massAssetFileEditAction));
663
664 try
665 {
666 Database.StartAssetEditing();
667 massAssetFileEditAction.Invoke();
668 }
669 catch (Exception ex)
670 {
671 Debug.LogError($"Exception during BatchEditing: {ex.Message}");
672
673 if (rethrowExceptions)
674 throw ex; // re-throw to caller
675 }
676 finally
677 {
678 Database.StopAssetEditing();
679 }
680 }
