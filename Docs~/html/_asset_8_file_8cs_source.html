468 internal static Boolean CopyInternal(Path sourcePath, Path destinationPath, Boolean overwriteExisting)
469 {
470 ThrowIf.ArgumentIsNull(sourcePath, nameof(sourcePath));
471 ThrowIf.ArgumentIsNull(destinationPath, nameof(destinationPath));
472 ThrowIf.AssetPathNotInDatabase(sourcePath);
473 ThrowIf.SourceAndDestPathAreEqual(sourcePath, destinationPath);
474
475 destinationPath.CreateFolders();
476
477 var success = AssetDatabase.CopyAsset(sourcePath, destinationPath);
478 SetLastErrorMessage(success ? String.Empty : $"failed to copy {sourcePath} to {destinationPath}");
479 return success;
480 }

696 public static void BatchEditing(Action massAssetFileEditAction, Boolean rethrowExceptions = false)
697 {
698 ThrowIf.ArgumentIsNull(massAssetFileEditAction, nameof(massAssetFileEditAction));
699
700 try
701 {
702 Database.StartAssetEditing();
703 massAssetFileEditAction.Invoke();
704 }
705 catch (Exception ex)
706 {
707 Debug.LogError($"Exception during BatchEditing: {ex.Message}");
708
709 if (rethrowExceptions)
710 throw ex; // re-throw to caller
711 }
712 finally
713 {
714 Database.StopAssetEditing();
715 }
716 }
